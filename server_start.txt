@echo off

REM Change to server bin folder
cd /d "C:\Samisti\CMS_SERVER\bin" || (
    echo ERROR: Cannot change directory! Check the path.
    pause
    exit /b
)

REM Ensure logs folder exists
if not exist "C:\Samisti\CMS_SERVER\logs" (
    mkdir "C:\Samisti\CMS_SERVER\logs"
)

REM Define log file (overwrite each time)
set "LOGFILE=C:\Samisti\CMS_SERVER\logs\server.log"

REM Start Node.js server in a new minimized window and detach
echo Starting Node server... Logs are in %LOGFILE%
start "" /min cmd /c "node www >> "%LOGFILE%" 2>&1"

REM Close this terminal immediately
exit
